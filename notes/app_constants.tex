\begin{align*}

Mass of a neutron
\begin{align*}
  m &= 1.0086649 \; \text{u} \\
    &= 1.67493 \times 10^{-27} \; \text{kg} \\
    &= 939.57 \; \frac{\text{MeV}}{c^2}
\end{align*}

\begin{align*}

\end{align*}

\begin{align*}
  h &= 6.626070040 \times 10^{-34} \; \text{J s} \\
    &= 4.135667662 \times 10^{-15} \; \text{eV s} \\
  \hbar &= \frac{h}{2\pi}
\end{align*}
